                mpRank: window.GAME_STATE.mpRank,
                lastUpdated: serverTimestamp()
            };
            try {
                await setDoc(getUserDataDocRef(userId), data, { merge: true });

            }

                    window.GAME_STATE.unlockedOperators = Object.keys(window.GAME_STATE.OPERATORS).filter(k => !window.GAME_STATE.OPERATORS[k].locked);

        // --- LLM Powered Lore Generation Feature (Simplified for space) ---
        async function retryOperation(operation, maxRetries, delay = 1000) {
            for (let i = 0; i < maxRetries; i++) {
                try { return await operation(); } 
                catch (error) {
                    if (i === maxRetries - 1) throw error;
                    await new Promise(resolve => setTimeout(resolve, delay * Math.pow(2, i)));
                }
            }
        }

        window.generateLore = async () => {
            if (!window.userId || !window.isAuthReady) {
                alertBox.show('Authentication Required', 'Please ensure you are logged in before accessing Chronos AI.', 'text-red-500');
                return;
            }
            const operator = window.GAME_STATE.OPERATORS[window.GAME_STATE.selectedOperator];
            if (!operator) return;
            const loreOutput = document.getElementById('lore-output');
            const loreButton = document.getElementById('lore-button');
            loreOutput.innerHTML = '<div class="text-center text-neon-text mt-4">Generating Chronos File... <i class="fas fa-spinner fa-spin"></i></div>';
            loreButton.disabled = true;

            const userQuery = `Write a gritty, post-apocalyptic backstory file (2-3 paragraphs, max 150 words) for a playable military operative in a game called 'Whiteout Skulls: Year 3012'. The operator is named '${operator.name}' and belongs to the '${operator.type}' class. They use a weapon called the ${window.WEAPONS[operator.weapon].name}.`;
            const systemPrompt = "You are the Lore Master for 'Whiteout Skulls 3012'. Your response must be highly thematic, dark, and futuristic, suitable for a mature FPS game. Do not use conversational language or headings; output only the story text. Use terms like 'Cryo-sleep', 'Wasteland', and 'Neo-Sector'.";

            const payload = {
                contents: [{ parts: [{ text: userQuery }] }],
                systemInstruction: { parts: [{ text: systemPrompt }] },
            };

            try {
                const response = await retryOperation(async () => {
                    const res = await fetch(GEMINI_API_URL, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });
                    if (!res.ok) throw new Error(`API call failed with status: ${res.status}`);
                    return res.json();
                }, MAX_RETRIES);

                const candidate = response.candidates?.[0];
                if (candidate && candidate.content?.parts?.[0]?.text) {
                    const text = candidate.content.parts[0].text;
                    let outputHTML = `<div class="p-4 border-l-4 border-neon-text bg-gray-900/50 mb-3 text-white">${text.replace(/\n/g, '<br><br>')}</div>`;
                    loreOutput.innerHTML = outputHTML;
                } else {
                    loreOutput.innerHTML = '<p class="text-neon-red">Error: AI failed to generate lore file. Try again.</p>';
                }
            } catch (error) {
                console.error("Gemini API Error:", error);
                loreOutput.innerHTML = '<p class="text-neon-red">Connection failed. Chronos AI is currently offline.</p>';
            } finally {
                loreButton.disabled = false;
            }
        };

        // Global Alert Box (to replace alert())
        const alertBox = {
            show: (title, message, colorClass) => {
                const modal = document.getElementById('custom-modal');
                document.getElementById('modal-title').textContent = title;

                document.getElementById('modal-title').className = `text-2xl font-bold mb-2 ${colorClass || 'text-white'}`;
                modal.classList.remove('opacity-0', 'pointer-events-none');
            },
            hide: () => {
                const modal = document.getElementById('custom-modal');
                modal.classList.add('opacity-0', 'pointer-events-none');
            }
        };
        window.alertBox = alertBox;
        
    </script>

        let canvas, ctx;
        let animationFrameId;

        // 2D Game Constants
        const TILE_SIZE = 40;
        const MAP_WIDTH = 800;
        const MAP_HEIGHT = 600;
        const PLAYER_SPEED = 4;
        const ENEMY_RADIUS = 15;
        const ENEMY_SPEED = 1;
        const ENEMY_MAX_HEALTH = 20;
        const ZOMBIE_COUNT_PER_WAVE = 5;

        // Game State Variables
        let player = { x: 0, y: 0, dx: 0, dy: 0, radius: 20, color: '#e94560', health: 100, weapon: {}, playerDesign: 'gingerbread' };
        let projectiles = [];
        let enemies = [];
        let kills = 0;
        let wave = 1;
        let mapData = []; 
        let currentMapTheme;
        let keys = {};
        let lastShotTime = 0;

        // --- Game Entity Classes ---

        class Projectile {
            constructor(x, y, angle, weaponStats) {
                this.x = x;
                this.y = y;
                this.radius = weaponStats.radius;
                this.color = weaponStats.color;
                this.damage = weaponStats.damage;
                this.velocity = {
                    x: Math.cos(angle) * weaponStats.speed,
                    y: Math.sin(angle) * weaponStats.speed
                };
            }

            draw() {
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2, false);
                ctx.fillStyle = this.color;
                ctx.fill();
            }

            update() {
                this.x += this.velocity.x;
                this.y += this.velocity.y;
                this.draw();
            }
        }

        class Enemy {
            constructor(x, y, radius, color, speed, maxHealth) {
                this.x = x;
                this.y = y;
                this.radius = radius;
                this.color = color;
                this.speed = speed;
                this.health = maxHealth;
            }

            draw() {
                ctx.save();
                ctx.translate(this.x, this.y);
                
                // Draw the Skull/Zombie Body
                ctx.fillStyle = this.color;
                // Main head
                ctx.beginPath();
                ctx.arc(0, 0, this.radius, 0, Math.PI * 2);
                ctx.fill();
                
                // Eyes (white)
                ctx.fillStyle = 'white';
                ctx.beginPath();
                ctx.arc(-this.radius * 0.35, -this.radius * 0.35, this.radius * 0.2, 0, Math.PI * 2);
                ctx.arc(this.radius * 0.35, -this.radius * 0.35, this.radius * 0.2, 0, Math.PI * 2);
                ctx.fill();
                
                // Inner eye (black pupil)
                ctx.fillStyle = 'black';
                ctx.beginPath();
                ctx.arc(-this.radius * 0.35, -this.radius * 0.35, this.radius * 0.1, 0, Math.PI * 2);
                ctx.arc(this.radius * 0.35, -this.radius * 0.35, this.radius * 0.1, 0, Math.PI * 2);
                ctx.fill();
                
                // Mouth (glowing red line)
                ctx.strokeStyle = '#ff0077';
                ctx.lineWidth = 2;
                ctx.beginPath();
                ctx.moveTo(-this.radius * 0.5, this.radius * 0.5);
                ctx.lineTo(this.radius * 0.5, this.radius * 0.5);
                ctx.stroke();

                ctx.restore();
                
                // Draw health bar above
                const barWidth = this.radius * 2;
                const barHeight = 4;
                const healthRatio = this.health / ENEMY_MAX_HEALTH;
                
                ctx.fillStyle = 'black';
                ctx.fillRect(this.x - this.radius, this.y - this.radius - barHeight - 2, barWidth, barHeight);
                
                ctx.fillStyle = healthRatio > 0.5 ? 'green' : (healthRatio > 0.2 ? 'yellow' : 'red');
                ctx.fillRect(this.x - this.radius, this.y - this.radius - barHeight - 2, barWidth * healthRatio, barHeight);
            }

            update(playerX, playerY) {
                // Simple AI: Move towards player
                const dx = playerX - this.x;
                const dy = playerY - this.y;
                const angle = Math.atan2(dy, dx);
                
                this.x += Math.cos(angle) * this.speed;
                this.y += Math.sin(angle) * this.speed;
                this.draw();
            }
        }

        // --- Character Drawing Function ---
        function drawShape(ctx, x, y, radius, color, designType) {
            ctx.save();
            // Translate is already done in drawPlayer, we draw relative to (0, 0)
            
            // Default body color
            ctx.fillStyle = color;

            switch (designType) {
                case 'gingerbread':
                    // Main body (Rounded square shape)
                    ctx.fillRect(-radius * 0.75, -radius * 1.5, radius * 1.5, radius * 3);
                    ctx.beginPath();
                    ctx.arc(0, 0, radius, 0, Math.PI * 2);
                    ctx.fill();
                    // Buttons (eyes/face)
                    ctx.fillStyle = 'white';
                    ctx.beginPath();
                    ctx.arc(-radius * 0.3, -radius * 0.3, radius * 0.15, 0, Math.PI * 2);
                    ctx.arc(radius * 0.3, -radius * 0.3, radius * 0.15, 0, Math.PI * 2);
                    ctx.fill();
                    break;
                case 'horns':
                    // Main Body (Circle)
                    ctx.beginPath();
                    ctx.arc(0, 0, radius, 0, Math.PI * 2);
                    ctx.fill();
                    // Horns (Triangles)
                    ctx.fillStyle = '#cc7700'; // Darker orange/brown
                    ctx.beginPath();
                    ctx.moveTo(-radius * 0.5, -radius * 1.5);
                    ctx.lineTo(-radius * 0.1, -radius * 0.5);
                    ctx.lineTo(-radius * 0.8, -radius * 0.5);
                    ctx.fill();
                    ctx.beginPath();
                    ctx.moveTo(radius * 0.5, -radius * 1.5);
                    ctx.lineTo(radius * 0.1, -radius * 0.5);
                    ctx.lineTo(radius * 0.8, -radius * 0.5);
                    ctx.fill();
                    break;
                case 'helmet':
                    // Main Body (Gray)
                    ctx.beginPath();
                    ctx.arc(0, 0, radius, 0, Math.PI * 2);
                    ctx.fill();
                    // Visor (Black rectangle)
                    ctx.fillStyle = 'black';
                    ctx.fillRect(-radius * 0.8, -radius * 0.3, radius * 1.6, radius * 0.6);
                    break;
                case 'cyborg':
                    // Main Body (Teal)
                    ctx.beginPath();
                    ctx.arc(0, 0, radius, 0, Math.PI * 2);
                    ctx.fill();
                    // Cybernetic eye (Glowing dot)
                    ctx.fillStyle = '#ff0077';
                    ctx.beginPath();
                    ctx.arc(radius * 0.5, -radius * 0.3, radius * 0.2, 0, Math.PI * 2);
                    ctx.fill();
                    break;
                case 'ghost':
                    // Main Body (White, slightly transparent)
                    ctx.fillStyle = 'rgba(211, 211, 211, 0.7)';
                    ctx.beginPath();
                    ctx.arc(0, 0, radius, 0, Math.PI * 2);
                    ctx.fill();
                    // Eye patch (Black)
                    ctx.fillStyle = 'black';
                    ctx.fillRect(0, -radius * 0.5, radius * 0.7, radius * 0.3);
                    break;
                case 'mutant':
                    // Main Body (Green, irregular shape - triangle)
                    ctx.beginPath();
                    ctx.moveTo(0, -radius * 1.2);
                    ctx.lineTo(radius * 1.2, radius * 0.8);
                    ctx.lineTo(-radius * 1.2, radius * 0.8);
                    ctx.closePath();
                    ctx.fill();
                    // Goo drip
                    ctx.fillStyle = '#32cd32';
                    ctx.fillRect(-radius * 0.5, radius * 0.8, radius * 1, radius * 0.5);
                    break;
                case 'fire':
                    // Main Body (Orange-Red, fiery shape - simple star)
                    ctx.beginPath();
                    ctx.moveTo(0, -radius * 1.5);
                    ctx.lineTo(radius * 0.5, -radius * 0.5);
                    ctx.lineTo(radius * 1.5, 0);
                    ctx.lineTo(radius * 0.5, radius * 0.5);
                    ctx.lineTo(0, radius * 1.5);
                    ctx.lineTo(-radius * 0.5, radius * 0.5);
                    ctx.lineTo(-radius * 1.5, 0);
                    ctx.lineTo(-radius * 0.5, -radius * 0.5);
                    ctx.closePath();
                    ctx.fill();
                    break;
                case 'arctic':
                    // Main Body (Blue, hooded shape - top flat)
                    ctx.beginPath();
                    ctx.arc(0, radius * 0.2, radius, 0, Math.PI * 2);
                    ctx.fill();
                    // Hood line (white)
                    ctx.fillStyle = 'white';
                    ctx.fillRect(-radius * 0.9, -radius * 0.7, radius * 1.8, radius * 0.3);
                    break;
                default:
                    // Fallback to simple circle
                    ctx.beginPath();
                    ctx.arc(0, 0, radius, 0, Math.PI * 2);
                    ctx.fill();
            }
            
            // Simple Weapon/Facing Indicator (All characters)
            ctx.strokeStyle = 'white';
            ctx.lineWidth = 2;
            ctx.beginPath();
            ctx.moveTo(0, 0);
            ctx.lineTo(radius + 5, 0);
            ctx.stroke();

            ctx.restore();
        }
        
        // --- Game Setup and Control Functions ---

        window.startGame = () => {
            if (window.GAME_STATE.isGameRunning) return;
            
            window.GAME_STATE.isGameRunning = true;
            document.getElementById('main-menu').classList.add('hidden');
            document.getElementById('game-screen').classList.remove('hidden');

            // Load player and weapon stats from selected operator
            const selectedOp = window.GAME_STATE.OPERATORS[window.GAME_STATE.selectedOperator];
            const weaponStats = window.WEAPONS[selectedOp.weapon];

            player = {
                x: MAP_WIDTH / 2, 
                y: MAP_HEIGHT / 2, 
                dx: 0, 
                dy: 0, 
                radius: selectedOp.playerRadius, 
                color: selectedOp.playerColor, 
                health: 100,
                weapon: weaponStats,
                playerDesign: selectedOp.playerDesign // Passed to drawing function
            };
            
            kills = 0;
            wave = 1;
            projectiles = [];
            enemies = [];
            
            // --- Crucial: Set the map theme based on current selection ---
            currentMapTheme = window.GAME_STATE.MAPS[window.GAME_STATE.currentMap]; 
            
            generateMap();
            spawnWave(wave);
            updateHUD();

            if (animationFrameId) cancelAnimationFrame(animationFrameId);
            animate();
        };
        
        window.endGame = () => {
            if (!window.GAME_STATE.isGameRunning) return;

            window.GAME_STATE.isGameRunning = false;
            document.getElementById('main-menu').classList.remove('hidden');
            document.getElementById('game-screen').classList.add('hidden');
            if (animationFrameId) cancelAnimationFrame(animationFrameId);
            
            const xpGained = kills * 15;
            const currencyGained = kills * 8;
            window.GAME_STATE.xp += xpGained;
            window.GAME_STATE.currency += currencyGained;
            window.GAME_STATE.level = Math.floor(window.GAME_STATE.xp / 1000) + 1; 
            window.saveGameData();

            alertBox.show('MISSION FAILED', `You were overwhelmed after Wave ${wave} with ${kills} kills. XP Gained: ${xpGained}, Currency Gained: ${currencyGained}.`, 'text-neon-red');
        };


        function init2D() {
            canvas = document.getElementById('game-canvas');
            ctx = canvas.getContext('2d');
            
            // Set canvas size (Fixed for this simple top-down game)
            canvas.width = MAP_WIDTH;
            canvas.height = MAP_HEIGHT;

            // Setup Input Listeners
            document.addEventListener('keydown', onKeyDown, false);
            document.addEventListener('keyup', onKeyUp, false);
            canvas.addEventListener('contextmenu', (event) => event.preventDefault()); // Block right-click menu
            canvas.addEventListener('mousedown', onMouseDown, false);
        }

        // --- Input Handlers ---
        function onKeyDown(event) {
            keys[event.key.toLowerCase()] = true;
            if (['w', 'a', 's', 'd'].includes(event.key.toLowerCase())) {
                 event.preventDefault(); 
            }
        }

        function onKeyUp(event) {
            keys[event.key.toLowerCase()] = false;
        }

        function onMouseDown(event) {
            if (!window.GAME_STATE.isGameRunning) return;
            
            // Right-click button code is 2
            if (event.button === 2) { 
                const now = Date.now();
                if (now - lastShotTime < player.weapon.cooldown) {
                    return; // Cooldown not finished
                }
                lastShotTime = now;

                const rect = canvas.getBoundingClientRect();
                const mouseX = event.clientX - rect.left;
                const mouseY = event.clientY - rect.top;
                
                const angle = Math.atan2(mouseY - player.y, mouseX - player.x);
                
                // Spawn projectile slightly offset from player center
                const spawnOffset = player.radius + 5;
                const startX = player.x + Math.cos(angle) * spawnOffset;
                const startY = player.y + Math.sin(angle) * spawnOffset;

                projectiles.push(new Projectile(startX, startY, angle, player.weapon));
            }
        }

        // --- Drawing and Game Logic ---

        function drawPlayer() {
            // Determine player rotation based on keys (if moving) or last movement
            let rotation = 0; // Default facing right (0 degrees)
            if (player.dx !== 0 || player.dy !== 0) {
                rotation = Math.atan2(player.dy, player.dx);
            }
            
            ctx.save();
            ctx.translate(player.x, player.y);
            ctx.rotate(rotation); 
            
            // Draw the custom shape relative to the rotated canvas center (0, 0)
            drawShape(ctx, 0, 0, player.radius, player.color, player.playerDesign);

            ctx.restore();

            // Draw a simple health indicator (outside rotation context)
            const healthRatio = player.health / 100;
            ctx.strokeStyle = healthRatio > 0.5 ? '#00ffc8' : (healthRatio > 0.2 ? '#ffcc00' : '#ff0077');
            ctx.lineWidth = 3;
            ctx.beginPath();
            ctx.arc(player.x, player.y, player.radius + 5, 0, Math.PI * 2 * healthRatio);
            ctx.stroke();
        }
        
        function generateMap() {
            mapData = [];
            const cols = MAP_WIDTH / TILE_SIZE;
            const rows = MAP_HEIGHT / TILE_SIZE;

            for (let r = 0; r < rows; r++) {
                mapData[r] = [];
                for (let c = 0; c < cols; c++) {
                    mapData[r][c] = (r % 2 === 0 && c % 2 === 0) || (r % 2 !== 0 && c % 2 !== 0) ? 0 : 1;
                }
            }
        }

        function drawMap() {
            const theme = currentMapTheme;
            ctx.fillStyle = theme.background;
            ctx.fillRect(0, 0, MAP_WIDTH, MAP_HEIGHT);

            const cols = MAP_WIDTH / TILE_SIZE;
            const rows = MAP_HEIGHT / TILE_SIZE;

            for (let r = 0; r < rows; r++) {
                for (let c = 0; c < cols; c++) {
                    const tileType = mapData[r][c];
                    ctx.fillStyle = tileType === 0 ? theme.tileMain : theme.tileAccent;
                    ctx.fillRect(c * TILE_SIZE, r * TILE_SIZE, TILE_SIZE, TILE_SIZE);
                    
                    ctx.strokeStyle = theme.background;
                    ctx.strokeRect(c * TILE_SIZE, r * TILE_SIZE, TILE_SIZE, TILE_SIZE);
                }
            }
        }

        function updatePlayerMovement() {
            player.dx = 0;
            player.dy = 0;

            if (keys['w']) player.dy = -PLAYER_SPEED;
            if (keys['s']) player.dy = PLAYER_SPEED;
            if (keys['a']) player.dx = -PLAYER_SPEED;
            if (keys['d']) player.dx = PLAYER_SPEED;

            // Normalize diagonal movement speed
            if (player.dx !== 0 && player.dy !== 0) {
                const speed = PLAYER_SPEED / Math.sqrt(2);
                player.dx = player.dx > 0 ? speed : (player.dx < 0 ? -speed : 0);
                player.dy = player.dy > 0 ? speed : (player.dy < 0 ? -speed : 0);
            }

            // Apply movement and boundary checks
            player.x += player.dx;
            player.y += player.dy;

            // Clamp player position to map boundaries
            player.x = Math.max(player.radius, Math.min(MAP_WIDTH - player.radius, player.x));
            player.y = Math.max(player.radius, Math.min(MAP_HEIGHT - player.radius, player.y));
        }
        
        function spawnEnemy() {
            let x, y;
            // Spawn around the edges
            if (Math.random() < 0.5) { // Spawn on X edges
                x = Math.random() < 0.5 ? -ENEMY_RADIUS : MAP_WIDTH + ENEMY_RADIUS;
                y = Math.random() * MAP_HEIGHT;
            } else { // Spawn on Y edges
                x = Math.random() * MAP_WIDTH;
                y = Math.random() < 0.5 ? -ENEMY_RADIUS : MAP_HEIGHT + ENEMY_RADIUS;
            }
            // All enemies are dark red and have the same health
            enemies.push(new Enemy(x, y, ENEMY_RADIUS, '#8B0000', ENEMY_SPEED, ENEMY_MAX_HEALTH));
        }

        function spawnWave(waveNumber) {
            const count = ZOMBIE_COUNT_PER_WAVE + (waveNumber * 2); 
            enemies = []; 
            
            // Spawn all enemies instantly for simplicity
            for (let i = 0; i < count; i++) {
                spawnEnemy();
            }
        }
        
        function updateEnemies() {
            // Player's coordinates for AI target
            const playerX = player.x;
            const playerY = player.y;

            // Update enemy positions and check player collision
            for (let i = enemies.length - 1; i >= 0; i--) {
                let enemy = enemies[i];
                
                // 1. Enemy AI and Drawing
                enemy.update(playerX, playerY); 

                // 2. Collision with Player
                const distPlayer = Math.hypot(playerX - enemy.x, playerY - enemy.y);
                if (distPlayer - player.radius - enemy.radius < 1) {
                    player.health -= 1; // Continuous damage on contact
                    if (player.health <= 0) {
                        window.endGame();
                        return; 
                    }
                }
            }
            
            // 3. Projectile Collision Check (reverse loop for safe removal)
            for (let pIndex = projectiles.length - 1; pIndex >= 0; pIndex--) {
                const projectile = projectiles[pIndex];
                let hit = false;
                
                for (let eIndex = enemies.length - 1; eIndex >= 0; eIndex--) {
                    const enemy = enemies[eIndex];
                    const distBullet = Math.hypot(projectile.x - enemy.x, projectile.y - enemy.y);

                    if (distBullet - enemy.radius - projectile.radius < 1) {
                        // Bullet hit enemy
                        enemy.health -= projectile.damage;
                        hit = true; 

                        // Enemy Death Logic
                        if (enemy.health <= 0) {
                            enemies.splice(eIndex, 1);
                            kills++;
                        }
                        break; // Only hit one enemy per projectile
                    }
                }

                // Remove projectile if it hit an enemy or went off-screen
                if (hit || 
                    projectile.x + projectile.radius < 0 || 
                    projectile.x - projectile.radius > MAP_WIDTH || 
                    projectile.y + projectile.radius < 0 || 
                    projectile.y - projectile.radius > MAP_HEIGHT) {
                    
                    projectiles.splice(pIndex, 1);
                }
            }


            // 4. Wave End Check
            if (enemies.length === 0 && window.GAME_STATE.isGameRunning) {
                wave++;
                spawnWave(wave);
                player.health = Math.min(100, player.health + 20); // Heal player slightly
                alertBox.show('Wave Complete', `Preparing for Wave ${wave}! Health restored.`, 'text-yellow-500');
            }
        }


        function updateHUD() {
             document.getElementById('hud-wave').textContent = wave;
             document.getElementById('hud-zombies').textContent = enemies.length;
             document.getElementById('hud-kills').textContent = kills;
             document.getElementById('hud-weapon').textContent = player.weapon.name;
        }
        
        function animate() {
            if (!window.GAME_STATE.isGameRunning) return;

            // 1. Clear Canvas
            ctx.clearRect(0, 0, MAP_WIDTH, MAP_HEIGHT);

            // 2. Update Logic
            updatePlayerMovement();
            updateEnemies(); 

            // 3. Drawing
            drawMap();
            
            // Draw Projectiles
            projectiles.forEach(p => p.update());

            // Draw Entities
            enemies.forEach(e => e.draw());
            drawPlayer();
            
            // 4. Update UI
            updateHUD();

            // 5. Loop
            animationFrameId = requestAnimationFrame(animate);
        }

        window.initGame = () => {
            init2D();
        };
        
        document.addEventListener('DOMContentLoaded', window.initGame);

    </script>
